'use strict'

/**
 * A string matches pattern `[a-z0-9.]+`. For example `service.a`, `service01`.
 *
 * @typedef {string} ServiceIdentity
 */

/**
 * @interface
 */
class Service {
    /**
     * Identity of this service. The identity is use by other services for
     * it's dependencies specification in attribute
     * {@link Service.dependencies}.
     *
     * @type {ServiceIdentity}
     */
    static get identity() {
        throw new Error('static get identity()')
    }

    /**
     * List of dependency service by identities. On initialization,
     * {@link Container} pass instances of service identities to
     * {@link Service.open}.
     *
     * @type {Array<ServiceIdentity>}
     */
    static get dependencies() {
        throw new Error('static get dependencies()')
    }

    /**
     * Create an instance of service. Can be synchronize or asynchronize
     * function.
     *
     * @throws {Error}
     */
    static open() {
        throw new Error('static get open()')
    }

    /**
     * Stop service, release resource such as memory, connections.
     * Can be synchronize or asynchronize function.
     */
    close() {}
}

/**
 * @typedef {Map<ServiceIdentity, Service>} ServiceMap
 */

class Container {
    /**
     * @private
     * @type {ServiceMap}
     */
    _serviceMap

    /**
     * Initialize by {@link Container.open}.
     *
     * @param {ServiceMap} serviceMap
     */
    constructor(serviceMap) {
        this._serviceMap = serviceMap
    }

    /**
     * @param {Service} config - A special service, provide configuration for
     * initialization other services.
     * @param {Array<Service>} serviceTypes - Services must be in order which
     * all dependency of service `n` stand before it.
     * @return {Promise<Container>}
     * @throws {TypeError}
     */
    static async open(config, serviceTypes) {
        Container._validateConfig(config)
        if (!Array.isArray(serviceTypes)) {
            throw new TypeError('expect an array: serviceTypes')
        }
        let serviceMap = new Map([
            [config.constructor.identity, config]
        ])
        try {
            for (let ServiceType of serviceTypes) {
                Container._validateServiceType(ServiceType)
                let dependencyServices = Container._getDependencyServices(
                    serviceMap, ServiceType
                )
                let service = await ServiceType.open(...dependencyServices)
                if ((service instanceof ServiceType) === false) {
                    throw new TypeError(
                        `expect an instance service: ${ServiceType.name}.open`
                    )
                }
                serviceMap.set(ServiceType.identity, service)
            }
        }
        catch (error) {
            await Container._closeServices(serviceMap)
            throw error
        }
        return new Container(serviceMap)
    }

    /**
     * Stop all services and container itself.
     */
    async close() {
        await Container._closeServices(this._serviceMap)
    }

    /**
     * Retrieve service by identity.
     *
     * @param {ServiceIdentity} identity
     * @return {Service}
     * @throws {Error}
     */
    get(identity) {
        let service = this._serviceMap.get(identity)
        if (!service) {
            throw Error(`service not found: ${identity}`)
        }
        return service
    }

    /**
     * Put a service to container, it must be initilize outside.
     *
     * @param {ServiceIdentity} identity
     * @param {Service} instance
     * @throws {Error}
     */
    set(identity, instance) {
        if (this._serviceMap.has(identity)) {
            throw Error(`service conflict: ${identity}`)
        }
        this._serviceMap.set(identity, instance)
    }

    /**
     * @private
     * @param {Service} config
     * @throws {TypeError}
     */
    static _validateConfig(config) {
        if (typeof(config) !== 'object' || config === null) {
            throw new TypeError('expect an object: config')
        }
        Container._validateServiceType(config.constructor)
    }

    /**
     * @private
     * @param {ServiceMap} serviceMap
     * @param {Service} ServiceType
     * @return {Array<Service>}
     * @throws {Error}
     */
    static _getDependencyServices(serviceMap, ServiceType) {
        return ServiceType.dependencies.map(serviceName => {
            let service = serviceMap.get(serviceName)
            if (service === undefined) {
                let message = `no dependency service: ${serviceName}; ` +
                    `require by ${ServiceType.name}`
                throw new Error(message)
            }
            return service
        })
    }

    /**
     * @private
     * @param {Map<string, Service>} serviceMap
     */
    static async _closeServices(serviceMap) {
        let hasError = false
        for (let service of serviceMap.values()) {
            try {
                await service.close()
            }
            catch (error) {
                let message = `closing service ${service.constructor.name}: ` +
                    error.message
                console.log(message)
                hasError = true
            }
        }
        serviceMap.clear()
        if (hasError) {
            throw new Error('closing services')
        }
    }

    /**
     * @private
     * @param {Service} ServiceType
     */
    static _validateServiceType(ServiceType) {
        Container._validateServiceIdentity(ServiceType)
        Container._validateServiceDependency(ServiceType)
        Container._validateServiceOpen(ServiceType)
        Container._validateServiceClose(ServiceType)
    }

    /**
     * @private
     * @param {Service} ServiceType
     * @throws {TypeError}
     */
    static _validateServiceIdentity(ServiceType) {
        if (
            !Container._hasProperty(ServiceType, 'identity') ||
            typeof(ServiceType.identity) !== 'string' ||
            /^[a-z0-9.]+$/.test(ServiceType.identity) === false
        ) {
            let message = 'expect a string pattern [a-z0-9.]+: ' +
                `${ServiceType.name}.identity`
            throw new TypeError(message)
        }
    }

    /**
     * @private
     * @param {Service} ServiceType
     * @throws {TypeError}
     */
    static _validateServiceDependency(ServiceType) {
        if (
            !Container._hasProperty(ServiceType, 'dependencies') ||
            !Array.isArray(ServiceType.dependencies)
        ) {
            throw new TypeError(
                `expect a string array: ${ServiceType.name}.dependencies`
            )
        }
        for (let identity of ServiceType.dependencies) {
            if (!Container._isIdentity(identity)) {
                throw new TypeError(
                    `expect a string array: ${ServiceType.name}.dependencies`
                )
            }
        }
    }

    /**
     * @private
     * @param {Service} ServiceType
     * @throws {TypeError}
     */
    static _validateServiceOpen(ServiceType) {
        if (
            !Container._hasProperty(ServiceType, 'open') ||
            typeof(ServiceType.open) !== 'function'
        ) {
            throw new TypeError(`expect a function: ${ServiceType.name}.open`)
        }
    }

    /**
     * @private
     * @param {Service} ServiceType
     * @throws {TypeError}
     */
    static _validateServiceClose(ServiceType) {
        let prototype = ServiceType.prototype
        if (
            !Container._hasProperty(prototype, 'close') ||
            typeof(prototype.close) !== 'function'
        ) {
            throw new TypeError(`expect a function: ${ServiceType.name}.close`)
        }
    }

    /**
     * @private
     * @param {any} object
     * @param {string} name
     * @return {boolean}
     */
    static _hasProperty(object, name) {
        return Object.prototype.hasOwnProperty.call(object, name)
    }

    /**
     * @private
     * @param {string} value
     * @return {boolean}
     */
    static _isIdentity(value) {
        return typeof(value) === 'string' &&
            /^[a-z0-9.]+$/.test(value) === true
    }
}

module.exports = {
    Container
}
